function handleScore(pts, label) {
  if (!game || game.winner !== null) return;
  const p = game.players[game.current];
  if (p.currentThrow.length >= 3) return;

  // Check-in
  if (!p.isIn) {
    const valid = game.inRule === 'Double' ? label.startsWith('D') || label === 'Bull' :
                  game.inRule === 'Triple' ? label.startsWith('T') : true;
    if (!valid) {
      p.currentThrow.push({ pts:0, label, type:'noin' });
      if (p.currentThrow.length === 3) advanceTurn();
      renderGame(); return;
    }
    p.isIn = true;
  }

  const newScore = p.score - pts;
  if (newScore < 0 || newScore === 1) {
    p.currentThrow.push({ pts:0, label, type:'bust' });
    if (p.currentThrow.length === 3) advanceTurn();
    renderGame(); return;
  }

  if (newScore === 0) {
    const validOut = game.outRule === 'Normal' ||
      (game.outRule === 'Double' && (label.startsWith('D') || label === 'Bull')) ||
      (game.outRule === 'Triple' && label.startsWith('T'));
    if (!validOut) {
      p.currentThrow.push({ pts:0, label, type:'bust' });
      if (p.currentThrow.length === 3) advanceTurn();
      renderGame(); return;
    }
    p.score = 0;
    p.currentThrow.push({ pts, label, type:'hit' });
    p.darts.push(...p.currentThrow);
    p.turns++;
    game.winner = game.current;
    renderGame();
    finishGame();
    return;
  }

  p.score = newScore;
  p.currentThrow.push({ pts, label, type:'hit' });
  p.darts.push({ pts, label, type:'hit' });
  if (p.currentThrow.length === 3) advanceTurn();
  renderGame();
}

function advanceTurn() {
  const p = game.players[game.current];
  p.turns++;
  p.currentThrow = [];
  game.current = game.current === 0 ? 1 : 0;
}

function recordMiss() { handleScore(0, 'Miss'); }

function undoLast() {
  if (!game || game.winner !== null) return;
  const p = game.players[game.current];
  if (p.currentThrow.length === 0) return;
  const last = p.currentThrow.pop();
  if (last.type === 'hit') {
    p.score += last.pts;
    p.darts.pop();
  }
  selectedSector = null;
  document.getElementById('mult-panel').style.display = 'none';
  renderGame();
}

function skipTurn() {
  if (!game || game.winner !== null) return;
  advanceTurn();
  selectedSector = null;
  document.getElementById('mult-panel').style.display = 'none';
  renderGame();
}

function finishGame() {
  const w = game.players[game.winner];
  const l = game.players[game.winner === 0 ? 1 : 0];
  const calcP = (p) => {
    const real = p.darts.filter(d => d.type === 'hit');
    const total = real.reduce((s,d) => s+d.pts, 0);
    const avg = p.turns > 0 ? Math.round(total/p.turns*10)/10 : 0;
    const checkout = p.darts.length > 0 ? p.darts[p.darts.length-1].label : '';
    return { avg, checkout };
  };
  const ws = calcP(w), ls = calcP(l);
  const rec = {
    date: new Date().toISOString(),
    startScore: game.startScore,
    winner: w.name, loser: l.name,
    winnerTurns: w.turns, loserTurns: l.turns,
    winnerAvg: ws.avg, loserAvg: ls.avg,
    winnerCheckout: ws.checkout,
  };
  stats.games.push(rec);
  [w.name, l.name].forEach(name => {
    if (!stats.players[name]) stats.players[name] = { wins:0, losses:0, totalAvg:0, gameCount:0, bestGame:null, checkouts:0 };
  });
  const wp = stats.players[w.name];
  wp.wins++; wp.totalAvg += ws.avg; wp.gameCount++; wp.checkouts++;
  if (!wp.bestGame || w.turns < wp.bestGame) wp.bestGame = w.turns;
  const lp = stats.players[l.name];
  lp.losses++; lp.totalAvg += ls.avg; lp.gameCount++;
  saveStats();
}

function renderGame() {

    const real = p.darts.filter(d=>d.type==='hit');
    const total = real.reduce((s,d)=>s+d.pts,0);
    const avg = p.turns>0 ? Math.round(total/p.turns*10)/10 : 0;
    const slots = [0,1,2].map(di => {
      const d = i===game.current ? p.currentThrow[di] : null;
      const cls = d ? (d.type==='bust'?'bust':d.type==='noin'?'noin':'hit') : '';
      const txt = d ? (d.type==='bust'?'BUST':d.type==='noin'?'NO IN':d.label) : '·';
      return `<div class="throw-slot ${cls}">${txt}</div>`;
    }).join('');
    const isActive = game.current===i && !game.winner;
    return `<div class="player-card ${isActive?'active-player':''}">
      <div class="p-name">${isActive?'▶ ':''}${p.name.toUpperCase()}</div>
      <div class="p-score ${p.score===0?'zero':''}">${p.score}</div>
      <div class="p-meta">${p.turns} Aufn. ${avg>0?'· Ø '+avg:''}</div>
      ${!p.isIn?`<div class="p-need-in">CHECK-IN nötig</div>`:''}
      <div class="throw-track">${slots}</div>
    </div>`;
  }).join('');

  // Checkout hint
  const cp = game.players[game.current];
  const hint = cp && cp.score <= 170 && CHECKOUT_HINTS[cp.score];
  document.getElementById('checkout-hint').innerHTML = hint && !game.winner
    ? `<div class="checkout-hint"><span style="color:var(--accent2)">💡</span><span style="color:var(--accent2)">Checkout:</span><strong style="margin-left:4px">${hint}</strong></div>`
    : '';

  // Winner
  if (game.winner !== null) {
    const w = game.players[game.winner];
    const co = w.darts.length > 0 ? w.darts[w.darts.length-1].label : '';
    document.getElementById('winner-area').innerHTML = `
      <div class="winner-banner">
        <div class="winner-trophy">🏆</div>
        <div class="winner-name">${w.name} GEWINNT!</div>
        <div class="winner-sub">${w.turns} Aufnahmen · Checkout: ${co}</div>
        <button class="new-game-btn" onclick="showSetup()">NEUES SPIEL</button>
      </div>`;
    document.getElementById('dartboard').style.opacity = '0.3';
    document.getElementById('mult-panel').style.display = 'none';
  } else {
    document.getElementById('dartboard').style.opacity = '1';
  }
}

function showSetup() {
  document.querySelectorAll('.tab-btn').forEach((b,i) => b.classList.toggle('active', i===0));
  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
  document.getElementById('screen-setup').classList.add('active');
}

// SVG DARTBOARD
// ═══════════════════════════════════════════════════════════════
function drawBoard() {
  const svg = document.getElementById('dartboard');
  const cx=160, cy=160;
  const rings = [
    { r1:22,  r2:90,  type:'single'  },
    { r1:90,  r2:105, type:'triple'  },
    { r1:105, r2:130, type:'single2' },
    { r1:130, r2:145, type:'double'  },
  ];
  const evenColors = { single:'#10101e', triple:'#00c97a', single2:'#10101e', double:'#00c97a' };
  const oddColors  = { single:'#e8e8f0', triple:'#ff3d5a', single2:'#e8e8f0', double:'#ff3d5a' };
  let html = `<defs>
    <filter id="glow"><feGaussianBlur stdDeviation="3" result="b"/><feMerge><feMergeNode in="b"/><feMergeNode in="SourceGraphic"/></feMerge></filter>
    <radialGradient id="bgGrad" cx="50%" cy="50%"><stop offset="0%" stop-color="#1a1a30"/><stop offset="100%" stop-color="#08080f"/></radialGradient>
  </defs>
  <circle cx="${cx}" cy="${cy}" r="158" fill="#050508"/>
  <circle cx="${cx}" cy="${cy}" r="150" fill="url(#bgGrad)"/>`;

  SECTORS.forEach((num, i) => {
    const slice = 360/20;
    const sa = i*slice - slice/2, ea = sa+slice;
    rings.forEach(ring => {
      const even = i%2===0;
      const fill = even ? evenColors[ring.type] : oddColors[ring.type];
      const path = arcPath(cx,cy,ring.r1,ring.r2,sa,ea);
      const clickType = ring.type==='triple'?'T':ring.type==='double'?'D':'S';
      html += `<path d="${path}" fill="${fill}" stroke="#08080f" stroke-width="0.6"
        onclick="boardClick(${num},'${clickType}')" style="cursor:pointer" id="seg-${num}-${clickType}"/>`;
    });
  });

  // Number labels (between double and outer edge)
  SECTORS.forEach((num, i) => {
    const slice = 360/20;
    const angle = i*slice;
    const [lx,ly] = polar(cx,cy,angle,138);
    html += `<text x="${lx}" y="${ly}" text-anchor="middle" dominant-baseline="middle"
      fill="#fff" font-size="8.5" font-weight="700" font-family="'Share Tech Mono',monospace"
      style="pointer-events:none;user-select:none">${num}</text>`;
  });

  // Outer bull (25)
  html += `<circle cx="${cx}" cy="${cy}" r="22" fill="#ff3d5a" stroke="#08080f" stroke-width="0.8"
    onclick="boardClick(null,'BULL50')" style="cursor:pointer"/>`;
  // Inner bull (50)
  html += `<circle cx="${cx}" cy="${cy}" r="12" fill="#e8c84a" stroke="#08080f" stroke-width="0.8"
    onclick="boardClick(null,'BULL')" style="cursor:pointer"/>`;
  html += `<text x="${cx}" y="${cy}" text-anchor="middle" dominant-baseline="middle"
    fill="#000" font-size="6.5" font-weight="900" font-family="'Share Tech Mono',monospace"
    style="pointer-events:none">BULL</text>`;

  svg.innerHTML = html;
}

function polar(cx,cy,angleDeg,r) {
  const a = (angleDeg-90)*Math.PI/180;
  return [cx+r*Math.cos(a), cy+r*Math.sin(a)];
}
function arcPath(cx,cy,r1,r2,sa,ea) {
  const [x1,y1]=polar(cx,cy,sa,r1), [x2,y2]=polar(cx,cy,ea,r1);
  const [x3,y3]=polar(cx,cy,ea,r2), [x4,y4]=polar(cx,cy,sa,r2);
  const lf = (ea-sa)>180?1:0;
  return `M${x1} ${y1} A${r1} ${r1} 0 ${lf} 1 ${x2} ${y2} L${x3} ${y3} A${r2} ${r2} 0 ${lf} 0 ${x4} ${y4}Z`;
}

function boardClick(num, type) {
  if (!game || game.winner !== null) return;
  // Bulls
  if (type === 'BULL') { flashSeg(null,'BULL'); handleScore(50,'Bull'); return; }
  if (type === 'BULL50') { flashSeg(null,'BULL50'); handleScore(25,'Bull25'); return; }
  // Double/Triple — score immediately
  if (type === 'T') { flashSeg(num,'T'); handleScore(num*3,`T${num}`); return; }
  if (type === 'D') { flashSeg(num,'D'); handleScore(num*2,`D${num}`); return; }
  // Single — show multiplier panel
  selectedSector = num;
  showMultPanel(num);
}

function showMultPanel(num) {
  const panel = document.getElementById('mult-panel');
  panel.style.display = 'flex';
  panel.innerHTML = `
    <span class="mult-label">Sektor <strong>${num}</strong></span>
    <button class="mult-btn mult-s" onclick="confirmMult('S',${num})">
      <span>Single</span><span>S${num}</span><span class="pts">(${num})</span>
    </button>
    <button class="mult-btn mult-d" onclick="confirmMult('D',${num})">
      <span>Double</span><span>D${num}</span><span class="pts">(${num*2})</span>
    </button>
    <button class="mult-btn mult-t" onclick="confirmMult('T',${num})">
      <span>Triple</span><span>T${num}</span><span class="pts">(${num*3})</span>
    </button>
    <button class="mult-cancel" onclick="cancelMult()">✕</button>`;
}

function confirmMult(m, num) {
  const pts = m==='S'?num:m==='D'?num*2:num*3;
  flashSeg(num, m);
  document.getElementById('mult-panel').style.display = 'none';
  selectedSector = null;
  handleScore(pts, `${m}${num}`);
}

function cancelMult() {
  document.getElementById('mult-panel').style.display = 'none';
  selectedSector = null;
}

function flashSeg(num, type) {
  const id = type==='BULL'||type==='BULL50' ? null : `seg-${num}-${type==='S'?'S':type}`;
  if (!id) return;
  const el = document.getElementById(id);
  if (!el) return;
  const orig = el.getAttribute('fill');
  el.setAttribute('fill','#f0c040');
  setTimeout(() => el.setAttribute('fill', orig), 250);
}

// STATS RENDER
// ═══════════════════════════════════════════════════════════════
function renderStats() {
  const n = loadNames();
  const players = [n.p1, n.p2];
  const games = stats.games || [];
  const el = document.getElementById('stats-content');

  if (games.length === 0) {
    el.innerHTML = `<div class="empty-state"><div class="empty-icon">📊</div><div class="empty-text">NOCH KEINE SPIELE</div></div>`;
    return;
  }

  const getPS = (name) => {
    const p = stats.players?.[name] || {};
    const gc = (p.wins||0)+(p.losses||0);
    return {
      wins: p.wins||0, losses: p.losses||0, games: gc,
      avg: gc>0 ? Math.round(p.totalAvg/gc*10)/10 : 0,
      bestGame: p.bestGame||'–', checkouts: p.checkouts||0,
      wr: gc>0 ? Math.round((p.wins||0)/gc*100) : 0,
    };
  };

  const cards = players.map(name => {
    const s = getPS(name);
    return `<div class="stat-card">
      <div class="stat-card-name">${name.toUpperCase()}</div>
      <div class="stat-grid">
        ${[['Siege',s.wins,'var(--green)'],['Spiele',s.games,'var(--accent)'],
           ['Siegquote',s.wr+'%','var(--accent2)'],['Ø/Aufnahme',s.avg,'var(--text)'],
           ['Best Game',s.bestGame+' Auf.','var(--red)'],['Checkouts',s.checkouts,'var(--green)'],
        ].map(([l,v,c])=>`<div><div class="stat-item-label">${l}</div><div class="stat-item-val" style="color:${c}">${v}</div></div>`).join('')}
      </div>
    </div>`;
  }).join('');

  const recent = games.slice(-8).reverse().map(g => `
    <div class="recent-game">
      <div><div class="rg-winner">${g.winner}</div><div class="rg-loser">vs ${g.loser}</div></div>
      <div><div class="rg-meta">${g.startScore} · ${g.winnerTurns} Aufn.</div><div class="rg-date">${new Date(g.date).toLocaleDateString('de-DE')}</div></div>
    </div>`).join('');

  el.innerHTML = `
    <div class="stats-cards">${cards}</div>
    <div class="chart-card">
      <div class="chart-card-title">SIEGE VS NIEDERLAGEN</div>
      <canvas id="chartWins" height="130"></canvas>
    </div>
    <div class="chart-card">
      <div class="chart-card-title">Ø PRO AUFNAHME – VERLAUF</div>
      <canvas id="chartAvg" height="130"></canvas>
    </div>
    <div class="chart-card">
      <div class="chart-card-title">LETZTE SPIELE</div>
      ${recent}
    </div>
    <button class="pdf-btn" onclick="exportPDF()">📄 PDF EXPORTIEREN</button>`;

  // Build charts
  setTimeout(() => {
    if (chartWins) chartWins.destroy();
    if (chartAvg) chartAvg.destroy();

    const winsData = { labels: players, datasets: [
      { label:'Siege', data: players.map(n=>getPS(n).wins), backgroundColor:'#00c97a', borderRadius:5 },
      { label:'Niederlagen', data: players.map(n=>getPS(n).losses), backgroundColor:'#ff3d5a', borderRadius:5 },
    ]};
    chartWins = new Chart(document.getElementById('chartWins'), {
      type:'bar', data: winsData,
      options: { responsive:true, plugins:{ legend:{ labels:{ color:'#888', font:{family:'Share Tech Mono',size:10}}}},
        scales:{ x:{ticks:{color:'#888',font:{family:'Share Tech Mono',size:10}},grid:{color:'#1e1e40'}},
                 y:{ticks:{color:'#555'},grid:{color:'#1e1e40'}} }}
    });

    const last10 = games.slice(-10);
    const avgData = { labels: last10.map((_,i)=>`#${i+1}`), datasets: players.map((name,i) => ({
      label: name,
      data: last10.map(g => g.winner===name ? g.winnerAvg : g.loser===name ? g.loserAvg : null),
      borderColor: i===0?'#00d4ff':'#e8c84a', backgroundColor:'transparent',
      borderWidth:2, pointRadius:3, tension:0.3,
    }))};
    chartAvg = new Chart(document.getElementById('chartAvg'), {
      type:'line', data: avgData,
      options: { responsive:true, plugins:{ legend:{ labels:{ color:'#888', font:{family:'Share Tech Mono',size:10}}}},
        scales:{ x:{ticks:{color:'#555',font:{size:10}},grid:{color:'#1e1e40'}},
                 y:{ticks:{color:'#555'},grid:{color:'#1e1e40'}} }}
    });
  }, 50);
}

// PDF EXPORT
// ═══════════════════════════════════════════════════════════════
async function exportPDF() {
  const { jsPDF } = window.jspdf;
  const doc = new jsPDF({ unit:'mm', format:'a4' });
  const n = loadNames();
  const players = [n.p1, n.p2];
  const now = new Date();
  const dateStr = now.toLocaleDateString('de-DE', { day:'2-digit', month:'long', year:'numeric' });

  // ── Colors & helpers ──
  const C = { bg:[8,8,16], surface:[16,16,30], accent:[0,212,255], accent2:[232,200,74],
               green:[0,201,122], red:[255,61,90], text:[232,232,240], muted:[85,85,112] };

  const setFill = (rgb) => doc.setFillColor(...rgb);
  const setDraw = (rgb) => doc.setDrawColor(...rgb);
  const setTxt = (rgb)  => doc.setTextColor(...rgb);
  const W = 210, H = 297;

  // Background
  setFill(C.bg); doc.rect(0,0,W,H,'F');

  // Header bar
  setFill(C.surface); doc.rect(0,0,W,22,'F');
  setTxt(C.accent); doc.setFont('helvetica','bold'); doc.setFontSize(18);
  doc.text('🎯  DART STATISTIKEN', 14, 14);
  setTxt(C.muted); doc.setFont('helvetica','normal'); doc.setFontSize(9);
  doc.text(`Erstellt am ${dateStr}`, W-14, 14, { align:'right' });

  // Divider
  setDraw(C.accent); doc.setLineWidth(0.4); doc.line(14, 24, W-14, 24);

  let y = 32;

  // ── Player stat cards ──
  const getPS = (name) => {
    const p = stats.players?.[name] || {};
    const gc = (p.wins||0)+(p.losses||0);
    return { wins:p.wins||0, losses:p.losses||0, games:gc,
      avg: gc>0?Math.round(p.totalAvg/gc*10)/10:0,
      bestGame:p.bestGame||'–', checkouts:p.checkouts||0,
      wr: gc>0?Math.round((p.wins||0)/gc*100):0 };
  };

  const cardW = (W-14-14-8)/2;
  players.forEach((name, i) => {
    const s = getPS(name);
    const cx = 14 + i*(cardW+8);
    // Card bg
    setFill(C.surface); doc.roundedRect(cx, y, cardW, 52, 3, 3, 'F');
    setDraw(i===0?C.accent:C.accent2); doc.setLineWidth(0.3); doc.roundedRect(cx,y,cardW,52,3,3,'S');
    // Name
    setTxt(i===0?C.accent:C.accent2); doc.setFont('helvetica','bold'); doc.setFontSize(11);
    doc.text(name.toUpperCase(), cx+6, y+9);
    // Stats grid
    const items = [
      ['Siege', s.wins, C.green],['Spiele', s.games, C.accent],
      ['Siegquote', s.wr+'%', C.accent2],['Ø/Aufnahme', s.avg, C.text],
      ['Best Game', s.bestGame+' Auf.', C.red],['Checkouts', s.checkouts, C.green],
    ];
    items.forEach(([label, val, color], j) => {
      const col = j%2, row = Math.floor(j/2);
      const ix = cx+6 + col*(cardW/2);
      const iy = y+18 + row*14;
      setTxt(C.muted); doc.setFont('helvetica','normal'); doc.setFontSize(7);
      doc.text(label.toUpperCase(), ix, iy);
      setTxt(color); doc.setFont('helvetica','bold'); doc.setFontSize(13);
      doc.text(String(val), ix, iy+7);
    });
  });
  y += 60;

  // ── Charts via canvas snapshot ──
  const snapChart = async (canvasId) => {
    const canvas = document.getElementById(canvasId);
    if (!canvas) return null;
    return canvas.toDataURL('image/png');
  };

  // Wins chart
  const winsImg = await snapChart('chartWins');
  if (winsImg) {
    setFill(C.surface); doc.roundedRect(14, y, W-28, 58, 3,3,'F');
    setTxt(C.muted); doc.setFont('helvetica','bold'); doc.setFontSize(8);
    doc.text('SIEGE VS NIEDERLAGEN', 20, y+7);
    doc.addImage(winsImg,'PNG', 18, y+10, W-36, 44);
    y += 64;
  }

  // Avg chart
  const avgImg = await snapChart('chartAvg');
  if (avgImg) {
    setFill(C.surface); doc.roundedRect(14, y, W-28, 58, 3,3,'F');
    setTxt(C.muted); doc.setFont('helvetica','bold'); doc.setFontSize(8);
    doc.text('Ø PRO AUFNAHME – VERLAUF (LETZTE 10 SPIELE)', 20, y+7);
    doc.addImage(avgImg,'PNG', 18, y+10, W-36, 44);
    y += 64;
  }

  // ── Recent games table ──
  if (y < 240) {
    setTxt(C.muted); doc.setFont('helvetica','bold'); doc.setFontSize(8);
    doc.text('LETZTE SPIELE', 14, y+6); y+=10;
    // Header
    setFill(C.surface); doc.rect(14, y, W-28, 7, 'F');
    setTxt(C.accent); doc.setFont('helvetica','bold'); doc.setFontSize(7);
    ['Datum','Sieger','Verlierer','Modus','Aufnahmen','Ø Sieger'].forEach((h,i)=>{
      doc.text(h, 16+i*30, y+5);
    });
    y+=7;
    stats.games.slice(-8).reverse().forEach((g,i) => {
      setFill(i%2===0?C.bg:C.surface); doc.rect(14,y,W-28,7,'F');
      setTxt(C.text); doc.setFont('helvetica','normal'); doc.setFontSize(7);
      const row = [
        new Date(g.date).toLocaleDateString('de-DE'),
        g.winner, g.loser, `${g.startScore}`, `${g.winnerTurns}`, `${g.winnerAvg}`
      ];
      row.forEach((v,ci) => doc.text(v, 16+ci*30, y+5));
      y+=7;
    });
  }

  // Footer
  setFill(C.surface); doc.rect(0,H-12,W,12,'F');
  setTxt(C.muted); doc.setFont('helvetica','normal'); doc.setFontSize(7);
  doc.text('Dart Scorer App · Persönliche Statistiken', 14, H-4);
  doc.text(`Seite 1`, W-14, H-4, {align:'right'});

  doc.save(`dart-stats-${now.toISOString().slice(0,10)}.pdf`);
}
